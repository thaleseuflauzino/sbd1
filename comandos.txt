PRAGMA foreign_keys = ON;

CREATE TABLE Professor (
    cpf TEXT PRIMARY KEY, 
    nome TEXT NOT NULL,
    email TEXT NOT NULL,
    formacao TEXT NOT NULL
);

CREATE TABLE Disciplina (
    codigo TEXT PRIMARY KEY,
    nome TEXT NOT NULL,
    carga_horaria INTEGER NOT NULL,
    nome_curso TEXT NOT NULL,
    FOREIGN KEY (nome_curso) REFERENCES Curso(nome) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE Turma (
    id_turma INTEGER PRIMARY KEY,
    capacidade INTEGER NOT NULL
);

CREATE TABLE Ministra (
    id_turma INTEGER NOT NULL,
    codigo TEXT NOT NULL,
    cpf TEXT NOT NULL,
    PRIMARY KEY (id_turma, codigo),
    FOREIGN KEY (id_turma) REFERENCES Turma(id_turma)
   		ON DELETE RESTRICT
  		ON UPDATE CASCADE,
    FOREIGN KEY (codigo) REFERENCES Disciplina(codigo)
   		ON DELETE RESTRICT
  		ON UPDATE CASCADE,
    FOREIGN KEY (cpf) REFERENCES Professor(cpf)
   		ON DELETE RESTRICT
  		ON UPDATE CASCADE
);

CREATE TABLE Aluno (
    matricula INTEGER PRIMARY KEY,
    nome TEXT NOT NULL,
    status TEXT NOT NULL,
  	nome_curso TEXT NOT NULL,
  	FOREIGN KEY (nome_curso) REFERENCES Curso(nome) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE Monitora (
    matricula_aluno INTEGER NOT NULL,
    matricula_aluno_monitor INTEGER NOT NULL,
    PRIMARY KEY (matricula_aluno, matricula_aluno_monitor),
    FOREIGN KEY (matricula_aluno) REFERENCES Aluno(matricula)
  		ON DELETE CASCADE
  		ON UPDATE CASCADE,
    FOREIGN KEY (matricula_aluno_monitor) REFERENCES Aluno(matricula)
  		ON DELETE CASCADE
  		ON UPDATE CASCADE
);

CREATE TABLE Curso (
    nome TEXT PRIMARY KEY,
    carga_horaria INTEGER NOT NULL,
  	id_faculdade INTEGER NOT NULL,
  	FOREIGN KEY (id_faculdade) REFERENCES Faculdade(id)
  		On update cascade
  		on delete cascade
);

CREATE TABLE Faculdade (
    id INTEGER PRIMARY KEY,
    nome TEXT NOT NULL
);

CREATE TABLE Reitor (
    cpf TEXT PRIMARY KEY,
    nome TEXT NOT NULL,
    mandato INTEGER NOT NULL,
  	id_faculdade INTEGER NOT NULL,
    FOREIGN KEY (id_faculdade) REFERENCES Faculdade(id)
   		ON DELETE CASCADE
  		ON UPDATE CASCADE
);

CREATE TABLE Departamento (
   id_departamento INTEGER,
   id_faculdade INTEGER,
   quantidade_alunos INTEGER NOT NULL,
   sigla TEXT NOT NULL,
   PRIMARY KEY (id_faculdade, id_departamento),
   FOREIGN KEY (id_faculdade) REFERENCES Faculdade(id)
     ON DELETE CASCADE
     ON UPDATE CASCADE
);
